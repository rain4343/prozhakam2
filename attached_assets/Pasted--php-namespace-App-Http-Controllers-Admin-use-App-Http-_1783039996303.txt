<?php

namespace App\Http\Controllers\Admin;

use App\Http\Controllers\Controller;
use App\Models\Document;
use App\Models\DocumentLog;
use App\Models\Department;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Auth;
use Illuminate\Support\Facades\Storage;

class DocumentController extends Controller
{
    public function index()
    {
        // لێرەدا دەتوانین لۆجیکی ئاڵۆزتر دابنێین بۆ پیشاندانی نوسراوەکان بەپێی ڕۆڵ
        $documents = Document::with('creator')->latest()->paginate(15);
        return view('admin.documents.index', compact('documents'));
    }

    public function create()
    {
        return view('admin.documents.create');
    }

    public function store(Request $request)
    {
        $validated = $request->validate([
            'document_number' => 'required|string|unique:documents,document_number',
            'document_date' => 'required|date',
            'subject' => 'required|string|max:255',
            'attachment' => 'required|file|mimes:pdf|max:10240', // Max 10MB
        ]);

        $filePath = $request->file('attachment')->store('documents', 'public');

        $document = Document::create([
            'document_number' => $validated['document_number'],
            'document_date' => $validated['document_date'],
            'subject' => $validated['subject'],
            'file_path' => $filePath,
            'creator_id' => Auth::id(),
            'current_status' => 'دروستکرا',
        ]);

        DocumentLog::create([
            'document_id' => $document->id,
            'user_id' => Auth::id(),
            'action' => 'نوسراوەکە دروستکرا',
            'timestamp' => now(),
        ]);

        return redirect()->route('admin.documents.index')->with('success', 'نوسراوەکە بە سەرکەوتوویی دروستکرا.');
    }

    public function show(Document $document)
    {
        $document->load('creator', 'logs.user');
        $departments = Department::all();
        return view('admin.documents.show', compact('document', 'departments'));
    }

    public function forward(Request $request, Document $document)
    {
        // لێرەدا دەبێت permission check بکرێت
        // if (!Auth::user()->can('forward-document')) { abort(403); }

        $validated = $request->validate([
            'department_id' => 'required|exists:departments,id',
            'notes' => 'nullable|string',
        ]);

        $department = Department::find($validated['department_id']);

        $document->update(['current_status' => 'ئاڕاستەکرا بۆ ' . $department->name]);

        DocumentLog::create([
            'document_id' => $document->id,
            'user_id' => Auth::id(),
            'action' => 'نوسراوەکە ئاڕاستەکرا بۆ: ' . $department->name,
            'notes' => $validated['notes'],
            'timestamp' => now(),
        ]);

        return redirect()->route('admin.documents.show', $document)->with('success', 'نوسراوەکە بە سەرکەوتوویی ئاڕاستەکرا.');
    }
}
