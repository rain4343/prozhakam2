<?php

namespace App\Http\Controllers\Admin;

use App\Http\Controllers\Controller;
use App\Models\User;
use App\Models\Department;
use App\Models\Role;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Hash;

class UserController extends Controller
{
    public function index()
    {
        $users = User::with('department', 'roles')->latest()->paginate(10);
        return $users; // بۆ تاقیکردنەوەی کاتی
    }

    public function create()
    {
        $departments = Department::all();
        $roles = Role::all();
        // viewی فۆڕمی زیادکردن پیشان دەدات لەگەڵ لیستی هۆبە و ڕۆڵەکان
        return ['departments' => $departments, 'roles' => $roles];
    }

    public function store(Request $request)
    {
        $validatedData = $request->validate([
            'full_name' => 'required|string|max:150',
            'username' => 'required|string|max:50|unique:users,username',
            'password' => 'required|string|min:8|confirmed',
            'department_id' => 'required|exists:departments,id',
            'job_title' => 'nullable|string|max:100',
            'roles' => 'required|array', // دڵنیابە لەوەی ڕۆڵەکان وەک array دەنێردرێن
        ]);

        $validatedData['password'] = Hash::make($validatedData['password']);
        
        $user = User::create($validatedData);
        $user->roles()->sync($request->roles);

        return 'User created successfully!';
    }

    public function edit(User $user)
    {
        $departments = Department::all();
        $roles = Role::all();
        return [
            'user' => $user,
            'departments' => $departments,
            'roles' => $roles
        ];
    }

    public function update(Request $request, User $user)
    {
        $validatedData = $request->validate([
            'full_name' => 'required|string|max:150',
            'username' => 'required|string|max:50|unique:users,username,' . $user->id,
            'password' => 'nullable|string|min:8|confirmed',
            'department_id' => 'required|exists:departments,id',
            'job_title' => 'nullable|string|max:100',
            'roles' => 'required|array',
        ]);
        
        if (!empty($validatedData['password'])) {
            $validatedData['password'] = Hash::make($validatedData['password']);
        } else {
            unset($validatedData['password']);
        }

        $user->update($validatedData);
        $user->roles()->sync($request->roles);

        return 'User updated successfully!';
    }

    public function destroy(User $user)
    {
        // لۆجیکی پاراستنی ئەدمینی سەرەکی لێرەدا دادەنرێت
        $user->delete();
        return 'User deleted successfully!';
    }
}
